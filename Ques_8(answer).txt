<script>
  document.write("Assume there is a json object of this format var obj = { “id” : 4, “name” : “abc”,“id” : 10, “name” : “ab2”,“id” : 5, “name” : “abc3”,“id” : 6, “name” : “abc5 } Write a code to sort the object by id ?<br/><br/>");
  document.write("Answer : <br/><br/>");
  var array = [{
      "id": 4,
      "name": "abc"
  }, {
      "id": 10,
      "name": "ab2"
  }, {
      "id": 5,
      "name": "abc3"
  }, {
      "id": 6,
      "name": "abc5"
  }];
  //Compare the Function
  function GetSortOrder(prop) {
      return function(a, b) {
          if (a[prop] > b[prop]) {
              return 1;
          } else if (a[prop] < b[prop]) {
              return -1;
          }
          return 0;
      }
  }
//sort by using id
  array.sort(GetSortOrder("id"));
  document.write("Sorted By id : <br/>");
  for (var item in array) {
      document.write("<br/>id : " + array[item].id + ", " + "&nbsp;&nbsp;name : " + array[item].name);
  }
</script>
